│   ├── main.py           <-- 后端入口（整个项目的启动命令写在这里）
│   ├── agents/           <-- 智能体核心（大脑）
│   │   └── workflow.py   <-- LangGraph 工作流逻辑
│   ├── tools/            <-- 工具包（手和脚）
│   │   └── time_tool.py  <-- 查时间的工具
│   │   └── calc_tool.py  <-- 计算器工具
│   │   └── llm_tool.py   <-- 大语言模型工具
│   └── config/           <-- 配置文件（身份证）
│       └── settings.py   <-- 存放 API Key 和 模型名称
│
├── frontend/             <-- 前端代码区（Streamlit）
│   └── web.py            <-- 网页界面代码（用户看到的聊天窗口）
│
├── venv/                 <-- 虚拟环境文件夹（不用管它，运行项目才会生成）
├── .gitignore            <-- Git 忽略文件（非常重要，告诉 Git 不要上传哪些文件）
├── requirements.txt      <-- 依赖清单（告诉队友：“装这些软件才能跑”）
